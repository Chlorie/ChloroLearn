370  void reshape(const DefaultableArrayShape& shape)
371  {
372  int automatic = -1;
373  size_t size = 1;
374  shape_.clear();
375  for (size_t i = 0; i < shape.size(); i++)
376  {
377  if (shape[i] <= 0)
378  {
379  // Automatic dimension
380  if (shape[i] == -1)
381  {
382  if (automatic == -1)
383  automatic = i;
384  else
385  throw IllegalArgumentException("Multiple automatic dimensions");
386  }
387  else
388  throw ArgumentOutOfRangeException("The lengths should be positive or -1 for automatic");
389  }
390  else
391  size *= shape[i];
392  shape_.push_back(shape[i]);
393  }
394  const size_t data_size = data_.size();
395  // No automatic dimension
396  if (automatic == -1)
397  {
398  if (size != data_size)
399  throw MismatchedSizesException("Sizes don't match");
400  }
401  else
402  {
403  if (data_size % size != 0)
404  throw IllegalArgumentException("Automatic dimension is not an integer");
405  shape_[automatic] = data_size / size;
406  }
407  }
